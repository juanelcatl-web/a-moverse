const CACHE_NAME = 'amoverse-v2'; const ASSETS = [ './amoverse-unificado.html', './manifest.json' ]; self.addEventListener('install', (e) => { e.waitUntil(caches.open(CACHE_NAME).then(cache => cache.addAll(ASSETS))); }); self.addEventListener('fetch', (e) => { e.respondWith(caches.match(e.request).then(res => res || fetch(e.request))); });
